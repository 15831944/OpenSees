    node [expr $i+1] [lindex $x $i] [lindex $y $i]
}

     {1 2} {5 4} {6 3}  {3 2} {4 1}};

    set n1 [lindex $conn $i 0]
    set n2 [lindex $conn $i 1]

    element truss [expr $i+1] $n1 $n2 $A 1
}
